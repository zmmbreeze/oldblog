jQuery源码解读之core篇

在1.3版本的时候就拜读过jQuery的源码，但是那时候的模块化不够好，代码的阅读很困难！进入1.4之后代码模块化好多了。具体的源码参考github页面：http://github.com/jquery/jquery/tree/master/src

昨天读完了core部分的代码，本文中会贴出我注释的源码（注释不全，稍有帮助）。

jQuery的核心结构

可以看成如下的代码：

[js]var jQuery = (function() {

// 定义一个jQuery的本地变量
// Define a local copy of jQuery
var jQuery = function( selector, context ) {
// jQuery对象实际上只是init构造器
// The jQuery object is actually just
// the init constructor ‘enhanced’
return new jQuery.fn.init( selector, context );
},

// 拷贝原本的jQuery对象，防止重写
// Map over jQuery in case of overwrite
_jQuery = window.jQuery,

// 拷贝原本的$对象，防止重写
// Map over the $ in case of overwrite
_$ = window.$,

jQuery.fn = jQuery.prototype = {
init: function( selector, context ) {
// 处理选择器等等情况
// 返回结果
},

// 选择符
// Start with an empty selector
selector: &quot;&quot;,

// 版本号
// The current version of jQuery being used
jquery: &quot;@VERSION&quot;,

// jQuery对象的大小
// The default length of a jQuery object is 0
length: 0

};

// 将init方法的prototype指向jQuery.fn即jQuery.prototype（very important！）
// Give the init function the jQuery prototype for later instantiation
jQuery.fn.init.prototype = jQuery.fn;

// 暴露jQuery对象给全局对象
// Expose jQuery to the global object
return window.jQuery = window.$ = jQuery;

})();[/js]

由上面的代码可知，我们平时获取的jQuery对象是一个类数组对象。它有length表示所存的dom的数量，也像数组那样存储dom对象。一般使用jQuery时，没有使用new操作符却返回了一个对象。原因就在于jQuery的方法返回了new jQuery.fn.init对象，而jQuery.fn.init.prototype对象就是jQuery.fn即jQuery.prototype。

DOM ready事件的实现

[js] bindReady: function() {
if ( readyBound ) {
return;
}

    readyBound = true;

    // 确保不会出现browser ready事件已经发生了但是还在调用
    //     $(document).ready()
    // Catch cases where $(document).ready() is called after the
    // browser event has already occurred.
    if ( document.readyState === &quot;complete&quot; ) {
        return jQuery.ready();
    }

    // Mozilla，Opera和webkit的nightlies版现在
    // 支持DOMContentLoaded事件
    // Mozilla, Opera and webkit nightlies currently
    // support this event
    if ( document.addEventListener ) {
        // Use the handy event callback
        document.addEventListener( &quot;DOMContentLoaded&quot;, DOMContentLoaded, false );

        //给window.onload事件绑定（这个事件发生时，dom必然已经ready了）
        // A fallback to window.onload, that will always work
        window.addEventListener( &quot;load&quot;, jQuery.ready, false );

    // 如果是IE的事件模型
    // If IE event model is used
    } else if ( document.attachEvent ) {
        // 确保在onload之前生效，可能会慢但对iframes也有效
        // ensure firing before onload,
        // maybe late but safe also for iframes
        document.attachEvent(&quot;onreadystatechange&quot;, DOMContentLoaded);

        // A fallback to window.onload, that will always work
        window.attachEvent( &quot;onload&quot;, jQuery.ready );

        // 如果是IE并且不是frame，继续检查看是否document ready了
        // If IE and not a frame
        // continually check to see if the document is ready
        var toplevel = false;

        try {
            // frameElement:
            // https://developer.mozilla.org/en/DOM/window.frameElement
            // 如果window是被嵌入的则返回如(such as &lt;iframe&gt; or &lt;object&gt;)
            // 之类的元素,
            // 如果window是在toplevel那么返回null.
            toplevel = window.frameElement == null;
        } catch(e) {}

        // documentElement:
                    // https://developer.mozilla.org/en/DOM/document.documentElement
        // 返回rootElement
        if ( document.documentElement.doScroll &amp;&amp; toplevel ) {
            doScrollCheck();
        }
    }
}[/js]

jQuery对DOM ready的实现在Firefox、Opera和Webkit浏览器上使用了DOMContentLoaded事件实现了ready，又使用load以保险。在IE下使用了onreadystatechange和onload做保险，但是真正的实现利用了doScroll方法。

[js]function doScrollCheck() {
if ( jQuery.isReady ) {
return;
}

try {
    // If IE is used, use the trick by Diego Perini
    // http://javascript.nwbox.com/IEContentLoaded/
    document.documentElement.doScroll(&quot;left&quot;);
} catch(e) {
    setTimeout( doScrollCheck, 1 );
    return;
}

// and execute any waiting functions
jQuery.ready();

}[/js]

doScroll(“left”)方法在IE下只能在dom树准备好之后执行，所以可以用来判断dom树是否ready。

解析JSON的实现

[js]
parseJSON: function( data ) {
// 必须是string
if ( typeof data !== "string" || !data ) {
return null;
}
// 移除头部和尾部的空白（否则IE无法处理）
// Make sure leading/trailing whitespace is removed (IE can’t handle it)
data = jQuery.trim( data );
// 确保是一个JSON字符串
// Make sure the incoming data is actual JSON
// Logic borrowed from http://json.org/json2.js
if (/^[],:{}\s]$/.test(data.replace(/\(?:["\\/bfnrt]|u[0-9a-fA-F]{4})/g, "@").replace(/"[^"\\n\r]"|true|false|null|-?\d+(?:.\d)?(?:[eE][+-]?\d+)?/g, "]").replace(/(?:^|:|,)(?:\s[)+/g, "")) ) {
// 尝试使用window.JSON.parse解析字符串
// Try to use the native JSON parser first
return window.JSON && window.JSON.parse ?
window.JSON.parse( data ) : (new Function("return " + data))();
} else {
jQuery.error( "Invalid JSON: " + data );
}
}
[/js]
