Prototype库面向对象的实现原理

Prototype库作为一个老牌js的库，其实现面向对象的方法很值得借鉴。先看一段Prototype库的使用代码：

// properties are directly passed to create method
var Person = Class.create({
  initialize: function(name) {
    this.name = name;
  },
  say: function(message) {
    return this.name + ‘: ‘ + message;
  }
});

// when subclassing, specify the class you want to inherit from
var Pirate = Class.create(Person, {
// redefine the speak method
say: function($super, message) {
return $super(message) + ‘, yarr!’;
}
});

var john = new Pirate(‘Long John’);
john.say(‘ahoy matey’);
// -> “Long John: ahoy matey, yarr!”

仔细分析其中的代码，先看Class.create函数部分。全部实现在：http://github.com/sstephenson/prototype/blob/master/src/lang/class.js

var Class = (function() {

// 有些版本的js不支持for循环遍历属性，再次判断是否不支持
var IS_DONTENUM_BUGGY = (function(){
for (var p in { toString: 1 }) {
// check actual property name, so that it works with augmented Object.prototype
if (p === ‘toString’) return false;
}
return true;
})();

function subclass() {};
function create() {
//查API得知：$A用于将类数组转换成真数组
var parent = null, properties = $A(arguments);
if (Object.isFunction(properties[0])) //如果第一位参数是函数即父类，则shift，并赋值给parent
parent = properties.shift();
//定义一个klass函数，默认执行其initialize方法
function klass() {
this.initialize.apply(this, arguments);
}
//扩展klass函数，为其添加方法addMethods
Object.extend(klass, Class.Methods);
//设置klass的函数属性superclass（父类）、subclasses（子类数组）
klass.superclass = parent;
klass.subclasses = [];

//如果有父类，设置继承关系
if (parent) {
  subclass.prototype = parent.prototype;//？
  klass.prototype = new subclass;
  parent.subclasses.push(klass);
}

//添加此类的属性与方法
for (var i = 0, length = properties.length; i &lt; length; i++)
  klass.addMethods(properties[i]);

//如果没有initialize方法，则设置为空函数
if (!klass.prototype.initialize)
  klass.prototype.initialize = Prototype.emptyFunction;
//设置prototype的constructor值为klass，防止在进行klass.prototype = new subclass时导致的constructor为parent
klass.prototype.constructor = klass;
return klass;

}

function addMethods(source) {
var ancestor = this.superclass && this.superclass.prototype,
properties = Object.keys(source); //以Array形式返回source对象的所有属性

// IE6 for循环不能迭代`toString`和`valueOf`方法 (包括其他`Object.prototype`内建方法),
// 如果不能迭代则强制拷贝.
// 不拷贝其他内建方法（Object.prototype.*）是因为效率
if (IS_DONTENUM_BUGGY) {
  if (source.toString != Object.prototype.toString)
    properties.push("toString");
  if (source.valueOf != Object.prototype.valueOf)
    properties.push("valueOf");
}

for (var i = 0, length = properties.length; i &lt; length; i++) {
  var property = properties[i], value = source[property]; //取得属性与值
  if (ancestor &amp;&amp; Object.isFunction(value) &amp;&amp;
      value.argumentNames()[0] == "$super") { //如果有父类，且属性的值是函数，且第一个参数是$super
    var method = value;
    value = (function(m) {
      return function() { return ancestor[m].apply(this, arguments); }; //调用父类相同名字的函数
    })(property).wrap(method); //包装

    value.valueOf = method.valueOf.bind(method);
    value.toString = method.toString.bind(method);
  }
  this.prototype[property] = value; //添加到klass.prototype上使其能被klass的对象访问到
}

return this;

}

return {
create: create,
Methods: {
addMethods: addMethods
}
};
})();
以上是本人加了中文注释之后的class.js的源码，如果你对javascript的原型链比较了解，那么理解不成问题！

注意代码中的第30行，为什么prototype库要特意新建一个subclass作为中间层呢？先看如下的图：

上图是Javascript的原型链探究一文中实现继承的原型链示例图。下图是prototype库实现继承的原型链示例图。

Son.prototype值不同，另一个区别是prototype库没有执行Father函数，而是执行了subclass这个空函数。为什么要这么设计呢？

要从属性和方法所存的地方说起！Javascript的原型链探究一文中属性和方法是存储在每一个新建的对象中的，它们在new操作之后被新建。如下代码：

function Father() {
    this.name = “father”;
}

而prototype库中，通过addMethod方法，将以对象方式传入（见37行）的属性与方法放置在Father.prototype对象上！也就是与如下代码类似：

function Father(){
    this.initialize.apply(this, arguments);
}
Father.prototype = Object.extend(Father.prototype,{ //扩展了Father.prototype
    this.name = “father”;
});

既然属性与方法是放置于Father.prototype上，那么就不需要执行Father函数了，以防函数中写了什么奇怪的代码，导致bug。测试代码（基于prototype库）：

var Father = Class.create({
    initialize:function() {
        console.log(“Father init”);
    }
});
var Son = Class.create(Father,{
    initialize:function() {
        console.log(“Son init”);
    }
});
var s = new Son; //只执行了Son init

但是prototype库的实现方式有一个缺点：

var Father = Class.create({
    initialize:function() {
        console.log(“Father init”);
    }
});
var Son = Class.create(Father,{
    initialize:function() {
        console.log(“Son init”);
    }
});
var s = new Son; //只执行了Son init
Father.prototype.name = “father”;
cosnole.log(s.name);//打印father

如上代码，父类后加的变量体现在了子类中，总感觉这样会导致bug。不过毕竟js不是纯面向对象的语言，按照一定的编码规则不轻易修改类的prototye的属性就不会有问题。

PS：但是这点与Java语言不一样，Java中子类初始化时都必须隐形的调用父类的默认初始化函数！如果没有访问父类默认构造函数的访问权限（如默认构造函数是private），那么必须显示的调用父类的其他构造函数！代码如下：

public class Father {
    private Father() {
    }

public Father(String name){
}

}
public class Son extends Father{
public Son() {
super(“sss”);
}
}
